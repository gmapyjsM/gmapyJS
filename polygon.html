Funkce createPolygon zobrazuje polygon. Funkci předáváme jako parametr vlastnosti polygonu známé z Google Maps API
JavaScript
 $(document).ready(function() {
    $("#mapa").Gmapy({
        zoom: 14,
        latitude: 49.814782471725145, 
        longitude: 18.268461227416992,
        markers: [{  
            latitude: 49.814782471725145, 
            longitude: 18.268461227416992  
        }]
    });
 $.Gmapy.createPolygon({
			color: "#00CC00",
			fillColor: 'transparent',
			fillOpacity:	0.0,
			weight: 	2,
			coords:	[{
					latitude: 58.092393, 
					longitude: 9.883691
				},{ 
					latitude: 54.281258,
					longitude: 13.487207
				},{ 
					latitude: 54.740475,
					longitude: 5.752832
			}]
 });
