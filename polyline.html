MAP
Popis
Funkce createPolyline zobrazuje křivku na základě zadaných latitude a longitude souřadnic. Funkci předáváme jako parametr vlastnosti křivky známé z Google Maps API
JavaScript
 $(document).ready(function() {
    $("#mapa").Gmapy({
        zoom: 14,
        latitude: 49.814782471725145, 
        longitude: 18.268461227416992,
        markers: [{  
            latitude: 49.814782471725145, 
            longitude: 18.268461227416992  
        }]
    });
 $.Gmapy.createPolyline({
			color: "#00CC00",
			opacity: 0.5,
			weight:	4,
			coords:	[{
					latitude: 55.841394,
					longitude: 48.99502
				},{ 
					latitude: 53.189575, 
					longitude: 50.313379
				},{ 
					latitude: 48.651052,
					longitude: 44.424707
			}]
});
